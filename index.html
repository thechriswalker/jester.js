jester.js - is a gesture framework

Currently doesn't support pinch except on iOS and needs the ability to unbund events, but heres the code for this example:

var $ = function(s){ return document.querySelector(s); }
jester.bind(
    $('#playground'),
    jester.scroll('vertical'),
    {
      start: function(){ $('#vertical').className = 'g'; },
       move: function(){ $('#vertical span').innerText = this.curY - this.srcY; },
        end: function(){ $('#vertical').className = 'r'; }
    }
);

jester.bind(
  $('#playground'),
    jester.scroll('horizontal'),
    {
      start: function(){ $('#horizontal').className = 'g'; },
       move: function(){ $('#horizontal span').innerText = this.curX - this.srcX; },
        end: function(){ $('#horizontal').className = 'r'; }
    }
);

jester.bind(
    $('#playground'),
    jester.pinch,
    {
      start: function(){ $('#pinch').className = 'g'; },
       move: function(){ $('#pinch span').innerText = this.scale.toFixed(2); },
        end: function(){ $('#pinch').className = 'r'; }
    }
);
